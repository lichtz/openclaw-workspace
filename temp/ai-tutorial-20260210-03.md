### 步骤 2：定义工具函数

```python
import requests

def get_weather(city):
    """获取城市天气"""
    api_key = "your_api_key"
    url = f"https://api.weather.com/data/2.5/weather?q={city}&appid={api_key}"
    response = requests.get(url)
    return response.json()
```

### 步骤 3：实现 Agent 逻辑

```python
class WeatherAgent:
    def __init__(self):
        self.tools = {
            "weather": get_weather
        }
    
    def run(self, query):
        # 解析用户意图
        if "天气" in query:
            city = self.extract_city(query)
            result = self.tools["weather"](city)
            return f"{city}天气：{result['weather'][0]['description']}"
        
        return "抱歉，我只能查询天气"
    
    def extract_city(self, query):
        # 简单城市提取
        words = query.split()
        for word in words:
            if word in ["北京", "上海", "深圳", "广州"]:
                return word
        return "北京"
```
